#trying to engineer some crazy features

bank_feat = bank %>%
  mutate(
    loyalty = Card_Category * (Months_on_book/12 * (Months_Inactive_12_mon + 1)) +
      (Total_Trans_Ct * Total_Trans_Amt)/(Total_Ct_Chng_Q4_Q1 + 1),
    
    life_circumstance = GenderM + Income_Category/(Dependent_count+1),

    l_financial_strain = log((Total_Revolving_Bal + Credit_Limit) *
                               (Avg_Open_To_Buy + 1)),

    l_activity = Total_Relationship_Count * (Months_on_book / (Months_Inactive_12_mon + 1)) * (Avg_Utilization_Ratio+1),

    l_stability = log(Card_Category + Months_on_book + Total_Revolving_Bal +
      (Total_Amt_Chng_Q4_Q1 * Total_Ct_Chng_Q4_Q1)),
    
    Churn = as.factor(Churn)
  )

bank_feat = bank_feat %>%
  select(loyalty, life_circumstance, l_financial_strain, l_activity, l_stability, Churn)
